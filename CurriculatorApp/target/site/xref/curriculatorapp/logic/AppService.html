20  public class AppService {
21  
22      private UserDao userdao;
23  
24      public AppService(UserDao userdao) {
25          this.userdao = userdao;  
26      }
27      
28      public boolean login(String username, String password) throws SQLException {
29          User user = (User) userdao.findByUsername(username);
30          if (user == null) {
31              return false;
32          }
33          
34          if (!user.checkPassword(password, user.getPassword())) {
35              return false;
36          }
37          
38          return true;
39      }
40  
41      public void createNewUser(String name, String username, String password) throws SQLException {
42          User newUser = new User(name, username);
43          newUser.setPassword(password);
44          String hashedPassword = newUser.getPassword();
45          userdao.createUser(name, username, hashedPassword);
46      }
47      
48      public void createNewUserTableIfNotExists() throws SQLException {
49          userdao.createNewUserTable();
50      }
